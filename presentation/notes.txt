### Parentheses

stack ghci GenParens.hs --package ghc --package ghc-source-gen --package ghc-paths


*GenParens> render $ var "f" @@ var "g" @@ var "h"
f g h
*GenParens> render $ (var "f" @@ var "g") @@ var "h"
f g h
*GenParens> render $ var "f" @@ (var "g" @@ var "h")
f (g h)
*GenParens> render $ var "f" @@ tuple [var "g", var "h"]
f (g, h)
*GenParens> render $ var "f" @@ par (var "g" @@ var "h")
f (g h)
